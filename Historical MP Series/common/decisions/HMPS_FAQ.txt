read_FAQ = {
	is_shown = {
		NOT = { has_variable = disabled_FAQ }
		is_player = yes
	}

	possible = {
		NOT = { has_variable = reading_FAQ }
	}

	when_taken = {
		trigger_event = {
			id = HMPS_FAQ.1
			popup = yes
		}
		set_variable = reading_FAQ
	}
}


remove_FAQ = {
	is_shown = {
		NOT = { has_variable = disabled_FAQ }
		is_player = yes
	}

	when_taken = {
		set_variable = disabled_FAQ
	}
}
