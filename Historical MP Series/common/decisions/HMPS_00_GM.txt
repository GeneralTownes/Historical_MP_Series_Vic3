gm_player = {
	is_shown = {
		NOT = { has_global_variable = player_gm }
		is_player = yes
		any_country = {
			is_player = yes
			NOT = { this = ROOT }
		}
	}

	when_taken = {
		set_global_variable = player_gm
		add_modifier = {
			name = game_master
		}
	}
}

GM_Self_Action = {
	is_shown = {
		has_modifier = game_master
		NOT = { has_global_variable = GM_taking_action }
	}
	
	when_taken = {
		set_global_variable = {
			name = GM_taking_action
			days = 31
		}
		trigger_event = {
			id = gm_actions.1
			days = 0
			popup = yes
		}
		set_variable = { 
			name = actionee
			days = 95
		}
    }
}
