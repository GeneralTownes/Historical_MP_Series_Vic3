grant_agri_tech = {
	is_shown = {
		is_subject = yes
		NOT = { has_technology_researched = pumpjacks }
		NOT = { has_variable = asked_for_agri_tech }
	}

	possible = {
		overlord = {
			has_technology_researched = pumpjacks
		}
	}

	when_taken = {
		if = {
			limit = {
				is_player = yes
			}
			overlord = {
				trigger_event = { id = agri_tech_transfer.1 days = 0 popup = yes }
			}
			set_variable = player_asked_for_agri_tech
		}
		else = {
			add_technology_researched = intensive_agriculture
			add_technology_researched = pumpjacks
		}
		set_variable = asked_for_agri_tech
	}

	ai_chance = {
		base = 100
	}
}
