HMPS_utility_je = {

	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	
	is_shown_when_inactive = {
        is_ai = yes
        OR = {
            AND = {
                has_global_variable = utility_tag_country
                OR = {
                    this = global_var:utility_tag_country
                    global_var:utility_tag_country = {
                        OR = {
                            is_country_alive = no
                            OR = {
                                is_ai = no
                                NOT = { # Human players eligible only if they conquered all the AIs
                                    any_country = {
                                        is_ai = yes
                                    }
                                }
                            }
                        }
                    }
                }
            }
            NOT = {
                has_global_variable = utility_tag_country
            }
        }
    }
    
    possible = {
        always = yes
    }
    
    immediate = {
        if = {
            limit = {
                OR = {
                    NOT = {
                        has_global_variable = utility_tag_country
                    }
                    global_var:utility_tag_country = {
                        OR = {
                            is_country_alive = no
                            OR = {
                                is_ai = no
                                NOT = {
                                    any_country = {
                                        is_ai = yes
                                    }
                                }
                            }
                        }
                    }
                }
            }
            set_global_variable = {
                name  = utility_tag_country
                value = this
            }
        }
    }
    
    invalid = {
        OR = {
			AND = {
				is_ai = no
				any_country = {
					is_ai = yes
				}
			}
            AND = {
                has_global_variable = utility_tag_country
                this != global_var:utility_tag_country
            }
        }
    }
	
	on_weekly_pulse = {
		effect = {
			every_state = {
				limit = {
					has_decree = decree_emergency_relief
					devastation > 0
				}
				state_region = { add_devastation = -0.70 }
			}
		}
	}
	
	weight = 0

    transferable = no
    
    can_revolution_inherit = no

    can_deactivate = yes
    
    should_be_pinned_by_default = no
}
