	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	
	group = je_group_historical_content
	
	is_shown_when_inactive = {
		exists = c:RUS
		THIS = c:RUS
		hidden_trigger = {
			NOT = { has_variable = alexander_reforms_var }
		}
    }
	
    possible = {
		ruler = {
			has_ideology = ideology:ideology_land_reformer
		}
    }
	
	immediate = {
		set_variable = alexander_reforms_var
		set_variable = {
			name = je_fix_bear_is_awake
			days = 1
		}	
		if = {
			limit = {
				game_date >= 1855.01.01
				ruler = {
					NOT = { has_ideology = ideology:ideology_land_reformer }
				}
				heir = {
					has_ideology = ideology:ideology_land_reformer
				}
			}
			ruler = {
				kill_character = yes
			}
		}
		#add_journal_entry = {
		#	type = je_bulgaria_liberated
		#}
		#add_journal_entry = {
		#	type = je_serbia_liberated
		#}
		#add_journal_entry = {
		#	type = je_albania_liberated
		#}
	}

	complete = {
		NOT = { has_law = law_type:law_traditionalism }
		#NOT = { has_law = law_type:law_serfdom }
		#NOT = { has_law = law_type:law_hereditary_bureaucrats }
		#literacy_rate > 0.40
		hidden_trigger = {
			NOT = { has_variable = je_fix_bear_is_awake }
		}
	}

	on_complete = {
		#Alexandergames?
		add_modifier = {
			name = the_bear_wakes
			months = very_long_modifier_time
			is_decaying = yes
		}
	}

	timeout = 5500

	on_timeout = {
		#AlexanderAssasinated omw to start a revolution aristocrats get more power
		add_modifier = {
			name = the_bear_coma
			months = very_long_modifier_time
			is_decaying = yes
		}
		set_variable = russia_reform_complete
	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
	can_revolution_inherit = no
}

je_decembrist_sympathizers = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	
	group = je_group_historical_content
	
	immediate = {
		set_variable = {
			name = decembrist_elimination
			value = 0
		}
	}
	
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		NOT = { has_law = law_type:law_no_police }
		OR = {
			has_law = law_type:law_censorship
			has_law = law_type:law_outlawed_dissent
		}
		NOT = {
			any_interest_group = {
				NOT = { has_ideology = ideology:ideology_liberal }
				is_marginal = no
				leader = {
					OR = {
						has_ideology = ideology:ideology_radical
						has_ideology = ideology:ideology_republican_leader
						has_ideology = ideology:ideology_reformer
					}
				}
			}
		}
		NOT = {
			any_scope_character = { 
				OR = {
					has_role = general
					has_role = admiral
					has_role = agitator
				}
				OR = {
					has_ideology = ideology:ideology_radical
					has_ideology = ideology:ideology_republican_leader
					has_ideology = ideology:ideology_reformer
				}
			}
		}
	}

	on_complete = {
		custom_tooltip = {
			text = liberals_fail_tt
		}
		add_modifier = {
			name = decembrist_suppression_success
			months = very_long_modifier_time
			is_decaying = yes
		}
	}

	fail = {
		OR = {
			NOT = { has_law = law_type:law_monarchy }
			NOT = { has_law = law_type:law_autocracy }
		}
	}
	
	on_fail = {
		add_modifier = {
			name = decembrist_suppression_failure
			months = very_long_modifier_time
			is_decaying = yes
		}
	}
	
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					NOT = { has_law = law_type:law_no_police }
					OR = {
						has_law = law_type:law_censorship
						has_law = law_type:law_outlawed_dissent
					}
				}
				change_variable = {
					name = decembrist_elimination
					add = 1
				}
			}
		}
	}
	
	timeout = 5475
	
	on_timeout = {
		add_modifier = {
			name = decembrist_suppression_failure
			months = very_long_modifier_time
			is_decaying = yes
		}
	}
	
	current_value = {
		value = root.var:decembrist_elimination
	}

	goal_add_value = {
		value = 60
	}
	
	progressbar = yes
	
	can_revolution_inherit = yes

	weight = 5000
	should_be_pinned_by_default = yes
}
