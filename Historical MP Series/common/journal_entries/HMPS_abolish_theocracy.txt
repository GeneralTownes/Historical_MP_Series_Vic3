# triggered by on_law_enactment_started
je_abolish_theocracy = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	
	group = je_group_internal_affairs

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					NOT = { has_law = law_type:law_theocracy }
					NOT = {
						any_civil_war = {
							civil_war_progress > 0.5
							is_civil_war_type = revolution
						}
					}
					NOT = {
						any_interest_group = {
							law_stance = {
								law = law_type:law_theocracy
								value > neutral
							}
							is_powerful = yes
						}
					}
				}
				change_variable = {
					name = abolishing_theocracy_var
					add = 1
				}
			}
		}
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		NOT = { has_law = law_type:law_theocracy }
		NOT = {
			any_civil_war = {
				civil_war_progress > 0.5
				is_civil_war_type = revolution
			}
		}
		NOT = {
			any_interest_group = {
				law_stance = {
					law = law_type:law_theocracy
					value > neutral
				}
				is_powerful = yes
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = 1848.8
		}
	}

	invalid = {
		NOR = {
			is_enacting_law = law_type:law_republic
			has_law = law_type:law_republic
		}
	}

	current_value = {
		value = root.var:abolishing_theocracy_var
	}

	goal_add_value = {
		add = 60
	}

	progressbar = yes

	weight = 100
	
	should_be_pinned_by_default = yes
}
