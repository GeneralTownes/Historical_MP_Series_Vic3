je_heavenly_kingdom_main = {
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	is_shown_when_inactive = {
		exists = c:CHI
		THIS = c:CHI
	}

	possible = {
		OR = {
			AND = {
				has_technology_researched = romanticism
				any_scope_state = {
					any_neighbouring_state = {
						is_treaty_port = yes
						owner = {
							religion = rel:protestant
						}
					}
					is_capital = no
				}
			}
			any_scope_state = {
				is_incorporated = yes
				any_scope_pop = {
					has_pop_religion = protestant
				}
				turmoil > 0.15
				is_capital = no
			}
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				OR = {
					AND = {
						is_incorporated = yes
						any_scope_pop = {
							has_pop_religion = protestant
						}
					}
					any_neighbouring_state = {
						is_treaty_port = yes
						owner = {
							religion = rel:protestant
						}
					}
				}
				NOT = { is_capital = yes }
			}
			add_modifier = {
				name = god_worshippers
				months = short_modifier_time
			}
		}
		trigger_event = {
			id = heavenly.2
		}
	}

	on_monthly_pulse = {
		random_events = {
			5 = 0
			1 = heavenly.3
		}
	}

	complete = {
		OR = {
			AND = {
				has_law = law_type:law_total_separation
				literacy_rate > 0.4
			}
			NOT = {
				any_scope_state = {
					has_modifier = god_worshippers
				}
			}
		}
	}
	
	on_complete = {
		ig:ig_intelligentsia = { 
			add_ideology = ideology_anti_clerical
		}
	}

	fail = {
		any_scope_state = {
			has_modifier = god_worshippers
			any_scope_pop = {
				has_pop_religion = protestant
				religion_accepted = no
			}
			count >= 4
		}
	}

	on_fail = {
		trigger_event = {
			id = heavenly.4
		}
		ig:ig_intelligentsia = { 
			add_ideology = ideology_anti_clerical
		}
	}

	weight = 10000
	should_be_pinned_by_default = yes
}

je_taiping = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	complete = {
		custom_tooltip = {
			text = taiping_destroyed
			hidden_trigger = {
				NOT = {
					exists = c:TPG
				}
			}
		}
	}

	immediate = {
		c:TPG = {
			save_scope_as = taiping_scope
		}
	}

	on_complete = {
		trigger_event = { id = heavenly.5 days = 0 }
		hidden_effect = {
			remove_modifier = taiping_taiping_rebellion
		}
	}

	fail = {
		exists = c:TPG
		has_truce_with = c:TPG
	}

	on_fail = {
		if = {
			limit = { has_journal_entry = je_warlord_china }
			post_notification = warlord_china_explosion_point_add
			change_variable = {
				name = china_warlord_explosion
				add = 1
			}
		}
		hidden_effect = {
			remove_modifier = taiping_taiping_rebellion
			c:TPG = {
				remove_modifier = taiping_taiping_rebellion
			}
		}
	}

	weight = 1000
	should_be_pinned_by_default = yes

	inheritable = no
}
