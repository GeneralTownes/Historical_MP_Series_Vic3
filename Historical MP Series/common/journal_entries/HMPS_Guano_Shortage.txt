je_mansur_slaving_empire = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"
	
	group = je_group_historical_content
	
	# You must have at least landed voting, right of assembly, and voting rights to accept it
	
    possible = {
		game_date > 1870.1.1
		has_variable = guano_producer
		any_scope_state = {
			has_active_building = buidling_guano_mine
		}
    }
	
	immediate = {
	}
	
	complete = {
	}
	
	fail = {
		custom_tooltip = {
			text = je_mansur_slaving_empire.t
			NOT = { exists = c:NUE }
			NOT = { exists = c:TPS }
			NOT = { exists = c:DNK }
			NOT = { exists = c:DFT }
			NOT = { owns_entire_state_region = STATE_EQUATORIA }
		}
	}
	
	on_monthly_pulse = {
		random_events = {
			11 = 0
			1 = mansur_slaving.1
		}
	}
	
	on_complete = {
		set_variable = mansur_empire
		add_modifier = {
			name = slaving_proceeds
			multiplier = 10000
			months = normal_modifier_time
		}
		random_scope_state = {
			limit = {
				state_region = s:STATE_EQUATORIA
			}
			add_modifier = {
				name = equatoria_conquered
				months = normal_modifier_time
			}
		}
	}
	
	progressbar = no
	weight = 10000
	should_be_pinned_by_default = yes

}
