namespace = prussian_germany
#Event to make Prussia get Alsace or all of Austria

prussian_germany.1 = {
	hidden = yes
	
	trigger = {
		NOT = { has_variable = prussian_germany }
		exists = c:PRU
		c:PRU = root
	}
	
	immediate = {
		set_variable = {
			name = prussian_germany
		}
	}
}


#I don't think there is a situation where this event happens, but just in case
prussian_germany.2 = {
	type = country_event
	placement = ROOT
	title = prussian_germany.2.t
	desc = prussian_germany.2.d
	flavor = prussian_germany.2.f
	
	popup = yes
	
	duration = 3
	
	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}
	
	trigger = {
		exists = c:GER
		c:GER = ROOT
		has_variable = prussian_germany
	}
	
	immediate = {
		remove_variable = prussian_germany
	}
	
	option = {
		remove_variable = prussian_germany
		add_journal_entry = {
			type = je_prussian_germany
		}
		add_loyalists = {
			value = 0.1
			culture = cu:north_german
		}
		add_radicals = {
			value = 0.25
			culture = cu:south_german
		}
	}
}
