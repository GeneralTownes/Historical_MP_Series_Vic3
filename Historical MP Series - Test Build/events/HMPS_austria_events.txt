namespace = austria_monarchy

#Flavour events for Austrian monarchs

austria_monarchy.1 = { #Abdication of Ferdinand I
	type = country_event
	placement = root

	title = austria_monarchy.1.t
	desc = austria_monarchy.1.d
	flavor = austria_monarchy.1.f

	gui_window = event_window_1char_tabloid
	
	left_icon = scope:aus_kaiser_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/law_icons/monarchy.dds"

	duration = 3

	cooldown = { months = stupidly_long_modifier_time }

	trigger = {
		has_variable = hungarian_revolt 
	}

	immediate = {
		ruler = {
			save_scope_as = aus_kaiser_scope
			save_scope_as = ruler_scope
		}
		root = {
			save_scope_as = austria_scope
		}
	}

	option = {
		name = austria_monarchy.1.a
		default_option = yes
		custom_tooltip = {
			text = retire_ruler
			ruler = {
				kill_character = { hidden = yes }
			}
		}
		add_radicals = {
			value = 0.05
			culture = cu:czech
		}
		add_loyalists = {
			value = 0.15
			culture = cu:south_german
		}
	}
	option = {
		name = austria_monarchy.1.b
		add_loyalists = {
			value = 0.05
			culture = cu:czech
		}
		add_radicals = {
			value = 0.15
			culture = cu:south_german
		}
		add_radicals = {
			value = 0.2
			culture = cu:hungarian
		}
		add_radicals = {
			value = 0.2
			culture = cu:north_italian
		}
	}
}
